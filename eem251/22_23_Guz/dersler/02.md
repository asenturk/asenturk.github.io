### BitRead komutu ikili sayıcı

![image](files/02/5.png)


```C++
int sayac=0;

void setup()
{
	for(int i=0;i<7;i++)
		pinMode(i, OUTPUT);  
} 
void loop()
{
	int bit_deger;
	for(int n=0;n<7;n++){
		bit_deger=bitRead(sayac, n);
		digitalWrite(n, bit_deger);
	} 
	sayac++;
	if(sayac==128)
		sayac=0;
}
```
