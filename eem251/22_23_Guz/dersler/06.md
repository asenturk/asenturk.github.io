Fotoresistör değeri 1.5k ise led yansın.

```C++
void setup(){
  pinMode(A2, INPUT);
  pinMode(6, OUTPUT);
}
void loop()
{
  int adc_degeri;
  float analog;
   adc_degeri= analogRead(A2);
   analog=(adc_degeri/1023.0)*5;

   if(analog>=3){
    digitalWrite(6,HIGH);
   }else{
    digitalWrite(6,LOW);
   }
  
  delay(100);
}
```

    digitalWrite(6,HIGH);
    aydinliktan_karanliga=0;
   }
   if(analog<=2.17 && aydinliktan_karanliga==0){
    digitalWrite(6,LOW);
    aydinliktan_karanliga=1;
   }
  
 
  delay(100);
}
