### Yapı işaretçisi
```C
#include <string.h>
struct calisan{
  char isim[30];
  int sicil;
  float maas;
};
int main(){
  struct calisan calisan1, *p;
  calisan1.maas=12500;
  calisan1.sicil=152;
  strcpy(calisan1.isim, "isim soyisim");
  p=&calisan1;
  
  (*p).maas = 15000;
  (*p).sicil =155;
  strcpy((*p).isim, "isim1 soyisim1");
  
  p->maas = 16000;
  p->sicil =160;
  strcpy(p->isim, "isim2 soyisim2");  
}
```
