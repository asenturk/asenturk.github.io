```C
#include <stdio.h>
//recursive fonksiyon ile us hesabı
//x^n hesabı
unsigned long us(int x, int n){
    if(n==1)
        return x;
    return x*us(x, n-1);
}
int main(){
    int taban, us_degeri;
    printf("taban ve us degerlerini giriniz: ");
    scanf("%d%d",&taban, &us_degeri);
    printf("%ld", us(taban, us_degeri));
    return 0;
}
```

